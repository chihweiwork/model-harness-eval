The implementation supports adding todos with auto-incrementing IDs, listing them with their completion status, and marking them as done while handling invalid IDs appropriately. ```python
import sys
import json
import os

STORAGE_FILE = 'todos.json'

def load_todos():
    if not os.path.exists(STORAGE_FILE):
        return []
    try:
        with open(STORAGE_FILE, 'r') as f:
            return json.load(f)
    except (json.JSONDecodeError, IOError):
        return []

def save_todos(todos):
    with open(STORAGE_FILE, 'w') as f:
        json.dump(todos, f)

def main():
    if len(sys.argv) < 2:
        return

    command = sys.argv[1]
    todos = load_todos()

    if command == 'add':
        if len(sys.argv) < 3:
            return
        text = " ".join(sys.argv[2:])
        next_id = 1
        if todos:
            next_id = max(t['id'] for t in todos) + 1
        
        todos.append({'id': next_id, 'text': text, 'done': False})
        save_todos(todos)
        print(f"added {next_id}")

    elif command == 'list':
        if not todos:
            print("no todos")
            return
        for t in todos:
            status = 'x' if t['done'] else ' '
            print(f"{t['id']} [{status}] {t['text']}")

    elif command == 'done':
        if len(sys.argv) < 3:
            return
        try:
            todo_id = int(sys.argv[2])
        except ValueError:
            print("not found")
            sys.exit(1)
            
        found = False
        for t in todos:
            if t['id'] == todo_id:
                t['done'] = True
                found = True
                break
        
        if found:
            save_todos(todos)
            print(f"done {todo_id}")
        else:
            print("not found")
            sys.exit(1)

if __name__ == "__main__":
    main()
